 * 3단계 - 오라클서버에게 처리를 요청
 *       a)select인 경우 - 커서를 조작해야함. 커서이동할 때 next() 호출-인스턴스화-클래스이름
 *         ResultSet rs = pstmt.executeQuery():ResulSet
 *       b)insert|update|delete - 커서가 필요 없다
 *         int result = pstmt.executeUpdate():int    
 **********************************************/

           public void getRentList(String pro_brand) {
                StringBuilder sql = new StringBuilder();
                sql.append("select member.mem_id,member.name      ");
                sql.append("       ,MEMBER.PHO_NO                 ");
               sql.append("   ,RENT.RENT_NO,RENT.ORDER_DATE      ");
               sql.append("   ,PRODUCT.Model_no,PRODUCT.CATEGORY ");
               sql.append(" from product,rent,member             ");
               sql.append("where MEMBER.MEM_ID=RENT.MEM_ID       ");
               sql.append(" and RENT.PRO_CODE=PRODUCT.PRO_CODE   ");
               try {
                Class.forName(RentOrclServer._DRIVER);
                Connection con =
                                DriverManager.getConnection(RentOrclServer._URL
                                                                  ,RentOrclServer._USER, RentOrclServer._PW);
                PreparedStatement pstmt = con.prepareStatement(sql.toString());
                ResultSet rs = pstmt.executeQuery();
                List<HashMap<String,Object>> list = new ArrayList<HashMap<String,Object>>();
                HashMap<String,Object> rmap = null;
                while(rs.next()) {
                      rmap = new HashMap<String,Object>();
                      rmap.put("mem_id",rs.getString("mem_id"));
                      rmap.put("name",rs.getString("name"));
                      rmap.put("PHO_NO",rs.getString("PHO_NO"));
                     rmap.put("RENT_NO",rs.getString("RENT_NO"));
                     rmap.put("ORDER_DATE",rs.getString("ORDER_DATE"));
                     rmap.put("Model_no",rs.getString("Model_no"));
                     rmap.put("CATEGORY",rs.getString("CATEGORY"));
                      list.add(rmap);
                }
                Iterator<HashMap<String,Object>> iter = list.iterator();
                Object obj[]=null;
                while(iter.hasNext()){
                      HashMap data = iter.next();
                      obj = data.keySet().toArray();
                      Vector oneRow= new Vector();
                      oneRow.add(data.get(obj[0]));
