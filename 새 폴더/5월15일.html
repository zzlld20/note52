import java.util.Random;
import java.util.Scanner;

public class For01 {

    public static void main(String[] args) {
        Random r = new Random();
        Scanner scan = new Scanner(System.in);
        int a;
        int b=0;
        int c=0;
        for(a=1;a<11;a++) {
            int i = r.nextInt(20);
            i=i-10;
            System.out.print(i);
            if(i>0) {
                b+=i;
            }
            else if(i<=0) {
                c+=i;
            }
        }
        System.out.println();
        System.out.println("양수의 합계는 : "+b);
        System.out.println("음수의 합계는 : "+c);
    }

}

    //import java.util.Random; 랜덤변수는 여기 산다
    public static void main(String[] args) {

        //r.nextInt(bound) bound 는 해당 숫자는 제외하고 뽑느다 -뒤에 있는건 소유주(r.하고 볼것)
        int i;
        int j;
        
        for(i=1;i<6;i++) {
            j=scan.nextInt();
            if(j==nansu) {
                System.out.println("정답입니다");
                break;
            }
            else if (nansu>j) {
                System.out.println("up");
                
            }
            else if (nansu<j) {
                System.out.println("down");
            }
            if(i==5) {
                System.out.println("바보");
            }
        }

}

}





package com.ch4;

import java.util.Random;
import java.util.Scanner;

public class game01 {

    public static void main(String[] args) {
        Random r = new Random();
        //컴퓨터가 채변한 숫자를 담을 변수 선언 = 채번한 숫자를 반환하는 함수호출
        int com = r.nextInt(10);
        //기회를 카운트해서 담을 변수 선언
        int cnt=0;
        Scanner scan = new Scanner(System.in);
        System.out.println("0~9 사이의 정수만 입력하세요");
        for(int i=0;i<5;i++) {
            
            int user=scan.nextInt();
            System.out.println("사용자가 입력한 숫자는 "+user);
            if(com==user) {//정답이라면
                System.out.println("정답입니다");
                System.exit(0);//가상머신과 연결을 끊어버린다.
            }
            else if(com<user) {//힌트는 낮춰라.
                System.out.println("입력한 숫자가 너무 크다. 낮춰라");
                cnt++;
                System.out.println("남은 기회는 "+(5-cnt));
            }
            else if(com>user) {//너무 작은 숫자를 입력할때
                
                if(cnt<5) {
                    System.out.println("입력한 숫자가 너무 낮다. 높여라");
                }
                cnt++;
                System.out.println("남은 기회는 "+(5-cnt));
            }
        }
            if(cnt==5) {
                System.out.println();
                System.out.println();
                System.out.println("넌 바보");
            }
        //for(int x=0;x<10;x++) {
        //int com = r.nextInt(10);
        //System.out.println(com);
        
    }

}
