package com.ch4;
import java.util.Scanner;
public class test02 {
     public static void main(String[] args) {
           int i;
           int j;
           int y;
           System.out.println("두개의 정수를 입력하세요");
           Scanner scan = new Scanner(System.in);
           i = scan.nextInt();
           j = scan.nextInt();
           if((j/3)-(i/3)>0) {
                y=(j/3)-(i/3);
                System.out.println("두개의 정수 사이의 3의배수 갯수는 "+y+"개 입니다");               
           }
           else if((i/3)-(j/3)>0) {
                y=(i/3)-(j/3);
                System.out.println("두개의 정수 사이의 3의배수 갯수는 "+y+"개 입니다");    
           }
           else {
                System.out.println("두 정수 사이에 3의 배수가 없습니다");
           }
           
     }
}
