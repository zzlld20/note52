package com.ch7;
public class TwoArray {
     int a[]=new int[3];//1차배열
     int a2[][]=new int[2][3];//2차배열
     String names[]= {"정지우","유승기","정재현"};
     int a3[][]= {
                     {70,80}
                   ,{90,75}
                   ,{85,60}
                     };
     //세 친구의 총점을 담을 1차 배열 선언하기
     int tal[]=new int[3];
     double avg[]=new double[tal.length];
     public void talInit() {
           for(int i=0;i<tal.length;i++) {
                tal[i]=a3[i][0]+a3[i][1];
                System.out.println(names[i]+"의 총점은 "+tal[i]+"점");
           }
           //avgsInit(a3[0].length);
     }
     //평균을 구해서 .avgs배열 초기화 하기 구현
     public void avgsInit(int inwon) {
           for(int i=0;i<avg.length;i++) {
                avg[i]=tal[i]/inwon;
                System.out.println(avg[i]);
           }               
     }
     ////석차를 구해서 출력해 보시오
     int tmp[]= {1,1,1};
     public void rank() {
           for(int i=0;i<tal.length;i++) {
                for(int j=i+1;j<tal.length;j++) {
                     if(tal[i]>tal[j]) {
                        tmp[j]++;
                                                
                     }
                }
           }
     }
     ///석차출력
     public void ranking() {
           //석차가 0등이 없기에 1부터 시작
           int rnks[]= {1,1,1};
           for(int i=0;i<rnks.length;i++) {
                for(int j=0;j<rnks.length;j++) {
                     //총점을 비교하여 총점이 작은 편이 순위를 1증가시킨다
                     if(tal[i]>tal[j]) {
                           rnks[j]++;
                     }
                }                    
           }
           for(int rnk:rnks) {
                System.out.println(rnk);
           }
     }
     
     public static void main(String[] args) {
           TwoArray ta = new TwoArray();
           System.out.println(ta.a[0]);//값이 출력된다
           System.out.println(ta.a2[0]);//주소번지가 출력된다 그림을 생각해볼것!
           System.out.println(ta.a2[0][0]);//값이 출력된다
           for(int i=0;i<ta.a2.length;i++) {//i<2
                for(int j=0;j<ta.a2[i].length;j++) {//j<3
                     System.out.println("a2["+i+"]["+j+"]="+ta.a2[i][j]);
                }
           }
           ta.talInit();
           ta.avgsInit(2);
           ta.ranking();
           
           
     }
}

package com.ch7;
import java.util.Random;
public class Test03 {
     int su[]=new int[100];
     int count[]=new int[10];
     
     public void methodA() {
           Random r = new Random();
           for(int i=0;i<su.length;i++) {
                su[i]=r.nextInt(10);
           }
     }
     public void methodB() {
           for(int j=0;j<count.length;j++) {
                for(int i=0;i<su.length;i++) {
                     if(su[i]==j) {
                           count[j]++;
                     }
                }
           }
     }
     public void methodC() {
           for(int i=0;i<count.length;i++) {
                System.out.println(i+"의 갯수는 :"+count[i]+"개");
           }
     }
     public static void main(String[] args) {
           Test03 t = new Test03();
           t.methodA();
           t.methodB();
           t.methodC();
     }
}
